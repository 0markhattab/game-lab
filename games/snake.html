Snake

The original mobile-phone classic. Move around a grid, eat the dot, grow longer. Run into yourself or the wall and it's over. Looks simple. The grid logic underneath is more interesting than you'd think.

How it works

Snake doesn't move every frame. It moves on a slow tick — about 8 times a second. Between ticks, the snake doesn't actually animate; it teleports one cell forward. Your eye fills in the motion.

The snake itself is just an array of {x, y} grid cells. To move forward: add a new head cell in the direction of travel, and remove the tail cell. To grow: skip removing the tail. That's the whole trick.

The classic Snake bug. If you press LEFT while moving RIGHT, the snake folds back on itself and dies on its own neck. We fix this by storing the direction as a "queued" turn that only commits on the next tick — and by rejecting any turn that's a 180° flip from the current heading.

// === SNAKE =====================================================
// Grid-based classic. The snake is an array of cells.
// Every "tick" we shift it forward by adding a new head and
// (usually) removing the tail.

const canvas = document.getElementById('game');
const ctx = canvas.getContext('2d');
const CELL = 20;
const COLS = canvas.width / CELL;   // 24
const ROWS = canvas.height / CELL;  // 24

let snake, dir, queuedDir, food, score, best = 0, alive, tickMs, lastTick;

function reset() {
  // Snake starts horizontal, 4 cells, in the middle, heading right
  snake = [
    { x: 10, y: 12 },
    { x: 11, y: 12 },
    { x: 12, y: 12 },
    { x: 13, y: 12 },
  ];
  dir = { x: 1, y: 0 };
  queuedDir = { x: 1, y: 0 };
  spawnFood();
  score = 0;
  alive = true;
  tickMs = 120;            // ~8 ticks/sec; speeds up as snake grows
  lastTick = performance.now();
}

function spawnFood() {
  // Pick a random cell that the snake doesn't occupy
  while (true) {
    const f = {
      x: Math.floor(Math.random() * COLS),
      y: Math.floor(Math.random() * ROWS)
    };
    if (!snake.some(s => s.x === f.x && s.y === f.y)) {
      food = f;
      return;
    }
  }
}

// --- Input: queue turns. Reject 180° flips. ---
window.addEventListener('keydown', e => {
  const map = {
    ArrowLeft:  { x:-1, y: 0 }, KeyA: { x:-1, y: 0 },
    ArrowRight: { x: 1, y: 0 }, KeyD: { x: 1, y: 0 },
    ArrowUp:    { x: 0, y:-1 }, KeyW: { x: 0, y:-1 },
    ArrowDown:  { x: 0, y: 1 }, KeyS: { x: 0, y: 1 },
  };
  const nd = map[e.code];
  if (nd) {
    e.preventDefault();
    // Block reversing onto your own neck
    if (nd.x !== -dir.x || nd.y !== -dir.y) queuedDir = nd;
  }
  if (e.code === 'Space' && !alive) reset();
});
document.getElementById('restart-btn').addEventListener('click', reset);

function tick() {
  // Commit the queued turn
  dir = queuedDir;

  // New head one cell in front
  const head = { x: snake[snake.length-1].x + dir.x,
                 y: snake[snake.length-1].y + dir.y };

  // Wall collision
  if (head.x < 0 || head.x >= COLS || head.y < 0 || head.y >= ROWS) {
    alive = false; return;
  }
  // Self collision (check every body cell — head about to enter it)
  if (snake.some(s => s.x === head.x && s.y === head.y)) {
    alive = false; return;
  }

  snake.push(head);

  // Eat food? Don't remove tail this tick = snake grows by 1
  if (head.x === food.x && head.y === food.y) {
    score++;
    if (score > best) best = score;
    spawnFood();
    // Speed up a touch every food. Don't go faster than 50ms/tick.
    tickMs = Math.max(50, tickMs - 3);
  } else {
    snake.shift();
  }
}

function draw() {
  // Background
  ctx.fillStyle = '#0a0e1a';
  ctx.fillRect(0, 0, canvas.width, canvas.height);

  // Grid lines (subtle blueprint feel)
  ctx.strokeStyle = 'rgba(35,44,74,0.6)';
  ctx.lineWidth = 1;
  for (let i = 0; i <= COLS; i++) {
    ctx.beginPath();
    ctx.moveTo(i * CELL, 0); ctx.lineTo(i * CELL, canvas.height);
    ctx.stroke();
    ctx.beginPath();
    ctx.moveTo(0, i * CELL); ctx.lineTo(canvas.width, i * CELL);
    ctx.stroke();
  }

  // Food (pulsing magenta)
  const pulse = 1 + Math.sin(performance.now() * 0.006) * 0.08;
  ctx.fillStyle = '#ff3eb5';
  ctx.shadowColor = '#ff3eb5';
  ctx.shadowBlur = 12;
  const fs = CELL * 0.7 * pulse;
  ctx.fillRect(food.x * CELL + (CELL - fs) / 2,
               food.y * CELL + (CELL - fs) / 2, fs, fs);
  ctx.shadowBlur = 0;

  // Snake — gradient from tail (dim) to head (bright)
  for (let i = 0; i < snake.length; i++) {
    const s = snake[i];
    const t = i / Math.max(1, snake.length - 1); // 0=tail, 1=head
    const isHead = i === snake.length - 1;
    if (isHead) {
      ctx.shadowColor = '#9eff5a';
      ctx.shadowBlur = 14;
    }
    ctx.fillStyle = `rgba(158,255,90,${0.4 + t * 0.6})`;
    ctx.fillRect(s.x * CELL + 1, s.y * CELL + 1, CELL - 2, CELL - 2);
    ctx.shadowBlur = 0;
  }

  // HUD
  ctx.fillStyle = '#9eff5a';
  ctx.font = 'bold 18px JetBrains Mono, monospace';
  ctx.textAlign = 'left';
  ctx.fillText(`SCORE ${score}`, 12, 24);
  ctx.font = '11px JetBrains Mono, monospace';
  ctx.fillStyle = '#9098b8';
  ctx.fillText(`BEST  ${best}`, 12, 40);

  // Death screen
  if (!alive) {
    ctx.fillStyle = 'rgba(255,90,106,0.25)';
    ctx.fillRect(0, 0, canvas.width, canvas.height);
    ctx.textAlign = 'center';
    ctx.fillStyle = '#fff';
    ctx.font = 'bold 36px Bricolage Grotesque, sans-serif';
    ctx.fillText('GAME OVER', canvas.width/2, canvas.height/2 - 6);
    ctx.font = '12px JetBrains Mono, monospace';
    ctx.fillStyle = '#9eff5a';
    ctx.fillText('press SPACE to restart', canvas.width/2, canvas.height/2 + 22);
  }
}

function loop(now) {
  // Two clocks: a slow tick (game logic) and a fast clock (drawing)
  if (alive && now - lastTick >= tickMs) {
    lastTick = now;
    tick();
  }
  draw();
  requestAnimationFrame(loop);
}

reset();
requestAnimationFrame(loop);
Things to try changing:
  • Make the walls wrap around — a snake leaving the right edge appears on the left.
  • Add a second food type: golden food worth 3 points but only stays for 5 seconds.
  • Slow the snake down instead of speeding it up. Which feels better?
  • Draw the snake as a smooth chain of circles instead of squares.
  • Add obstacles — fixed walls in the middle of the grid that you also can't touch.
