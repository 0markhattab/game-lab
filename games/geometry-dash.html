How it works

The cube doesn't actually move forward. It just bounces in place near the left edge of the screen. What moves is everything else — we slide the obstacles to the left by a value called scrollX that grows every frame. This is the same trick every side-scroller uses, from Mario to Geometry Dash.

game.js — Geometry Dash full source 
// === GEOMETRY DASH =============================================
// A scrolling rhythm-runner. One button. One life. Endless level.

const canvas = document.getElementById('game');
const ctx = canvas.getContext('2d');
const W = canvas.width;   // 800
const H = canvas.height;  // 360

// --- World constants ---
// Gravity is pixels-per-frame-per-frame. Jump impulse is a one-time
// velocity change. Tune these two numbers and the whole game feels
// different: heavy gravity = punchy jumps, light gravity = floaty.
const GRAVITY  = 0.7;
const JUMP_V   = -12.5;
const GROUND_Y = H - 50;
const BASE_SPEED = 4;

// --- Game state ---
let player, obstacles, particles, scrollX, speed, score, alive, deathT;
let best = 0;

function reset() {
  player    = { x: 110, y: GROUND_Y - 12, vy: 0, size: 24, rot: 0, onGround: true };
  obstacles = [];
  particles = [];
  scrollX   = 0;
  speed     = BASE_SPEED;
  score     = 0;
  alive     = true;
  deathT    = 0;
  // Pre-generate a stretch of obstacles so we never run out
  let nx = 600;
  while (nx < 5000) {
    obstacles.push({ x: nx, type: Math.random() < 0.28 ? 'double' : 'single' });
    nx += 200 + Math.random() * 280;
  }
}

// --- Input ---
// We capture jump intent in a flag. The actual jump happens in the
// loop only if the player is on the ground. This prevents floaty
// double-jumps unless you intentionally allow them.
let jumpRequested = false;
window.addEventListener('keydown', e => {
  if (e.code === 'Space' || e.code === 'ArrowUp') {
    e.preventDefault();
    if (alive) jumpRequested = true;
    else reset();
  }
  if (e.code === 'KeyR') reset();
});
canvas.addEventListener('mousedown', () => {
  if (alive) jumpRequested = true;
  else reset();
});
document.getElementById('restart-btn').addEventListener('click', reset);

// --- Collision: axis-aligned bounding box ---
// p is the player (x,y is center). box is x,y,w,h (top-left).
function hits(p, bx, by, bw, bh) {
  return p.x - p.size/2 < bx + bw &&
         p.x + p.size/2 > bx &&
         p.y - p.size/2 < by + bh &&
         p.y + p.size/2 > by;
}

// --- Death: explode into particles + a quick audio sting ---
function die() {
  alive = false;
  if (score > best) best = score;
  const colors = ['#ff3eb5', '#9eff5a', '#ffb347', '#5ce1ff'];
  for (let i = 0; i < 60; i++) {
    const a = (i / 60) * Math.PI * 2 + Math.random() * 0.3;
    const sp = 2 + Math.random() * 5;
    particles.push({
      x: player.x, y: player.y,
      vx: Math.cos(a) * sp, vy: Math.sin(a) * sp - 1,
      life: 40 + Math.random() * 25,
      maxLife: 60,
      color: colors[Math.floor(Math.random() * colors.length)],
      size: 2 + Math.random() * 3,
      gravity: true
    });
  }
  // Web Audio API: a short downward square-wave chirp = retro "dead"
  try {
    const ac = new (window.AudioContext || window.webkitAudioContext)();
    const o = ac.createOscillator();
    const g = ac.createGain();
    o.type = 'square';
    o.frequency.setValueAtTime(280, ac.currentTime);
    o.frequency.exponentialRampToValueAtTime(60, ac.currentTime + 0.35);
    g.gain.setValueAtTime(0.18, ac.currentTime);
    g.gain.exponentialRampToValueAtTime(0.001, ac.currentTime + 0.35);
    o.connect(g).connect(ac.destination);
    o.start();
    o.stop(ac.currentTime + 0.35);
  } catch(e) { /* audio is optional */ }
}

// --- Main loop ---
function loop() {
  // ============== UPDATE ==============
  if (alive) {
    // Speed ramps up with score for that "getting harder" feel
    speed = BASE_SPEED + Math.min(score / 1200, 4);
    scrollX += speed;
    score += speed * 0.5;

    // Player vertical physics
    player.vy += GRAVITY;
    player.y  += player.vy;
    if (player.y + player.size/2 >= GROUND_Y) {
      player.y = GROUND_Y - player.size/2;
      player.vy = 0;
      player.onGround = true;
      player.rot = 0;        // snap rotation when landing
    } else {
      player.onGround = false;
      player.rot += 0.20;    // spin while in air
    }

    // Trigger queued jump
    if (jumpRequested && player.onGround) {
      player.vy = JUMP_V;
      player.onGround = false;
    }

    // Trail particles
    if (Math.random() < 0.7) {
      particles.push({
        x: player.x - player.size/2 + (Math.random()-0.5) * 8,
        y: player.y + (Math.random()-0.5) * 8,
        vx: -1.5 - Math.random() * 1.5,
        vy: -0.3 + Math.random() * 0.6,
        life: 25 + Math.random() * 15,
        maxLife: 40,
        color: '#9eff5a',
        size: 2 + Math.random() * 1.5,
        gravity: false
      });
    }

    // Collide with each spike. Hitbox is slightly smaller than the
    // visual triangle to feel forgiving instead of cheap.
    for (const o of obstacles) {
      const ox = o.x - scrollX;
      if (ox > W + 50 || ox < -60) continue;
      if (hits(player, ox + 6, GROUND_Y - 24, 18, 24)) { die(); break; }
      if (o.type === 'double' &&
          hits(player, ox + 32, GROUND_Y - 24, 18, 24)) { die(); break; }
    }

    // Top up obstacles so the level never ends
    while (obstacles[obstacles.length-1].x - scrollX < 5000) {
      const last = obstacles[obstacles.length-1];
      obstacles.push({
        x: last.x + 200 + Math.random() * 280,
        type: Math.random() < 0.28 ? 'double' : 'single'
      });
    }
  } else {
    deathT++;
  }
  jumpRequested = false;

  // Update particles (always — they keep flying after death)
  for (let i = particles.length - 1; i >= 0; i--) {
    const p = particles[i];
    p.x += p.vx;
    p.y += p.vy;
    if (p.gravity) p.vy += 0.18;
    p.life--;
    if (p.life <= 0) particles.splice(i, 1);
  }

  // ============== DRAW ==============
  // Background — color subtly shifts with score
  const hue = (score * 0.05) % 360;
  const bg = ctx.createLinearGradient(0, 0, 0, H);
  bg.addColorStop(0, `hsl(${hue}, 35%, 9%)`);
  bg.addColorStop(1, '#0a0e1a');
  ctx.fillStyle = bg;
  ctx.fillRect(0, 0, W, H);

  // Parallax stars
  ctx.fillStyle = 'rgba(255,255,255,0.5)';
  for (let i = 0; i < 40; i++) {
    const sx = ((i * 173 - scrollX * 0.25) % W + W) % W;
    const sy = (i * 47) % (GROUND_Y - 20);
    ctx.fillRect(sx, sy, 1, 1);
  }

  // Ground line + hatching
  ctx.strokeStyle = '#9eff5a';
  ctx.lineWidth = 2;
  ctx.beginPath();
  ctx.moveTo(0, GROUND_Y);
  ctx.lineTo(W, GROUND_Y);
  ctx.stroke();

  ctx.strokeStyle = 'rgba(158,255,90,0.18)';
  ctx.lineWidth = 1;
  for (let x = -((scrollX * 1.5) % 32); x < W; x += 32) {
    ctx.beginPath();
    ctx.moveTo(x, GROUND_Y + 1);
    ctx.lineTo(x - 14, H);
    ctx.stroke();
  }

  // Particles (draw before player so trail sits behind cube)
  for (const p of particles) {
    ctx.globalAlpha = Math.min(1, p.life / 30);
    ctx.fillStyle = p.color;
    ctx.fillRect(p.x, p.y, p.size, p.size);
  }
  ctx.globalAlpha = 1;

  // Spikes
  ctx.fillStyle = '#ff3eb5';
  ctx.shadowColor = '#ff3eb5';
  ctx.shadowBlur = 8;
  for (const o of obstacles) {
    const ox = o.x - scrollX;
    if (ox > W + 50 || ox < -60) continue;
    ctx.beginPath();
    ctx.moveTo(ox, GROUND_Y);
    ctx.lineTo(ox + 15, GROUND_Y - 28);
    ctx.lineTo(ox + 30, GROUND_Y);
    ctx.fill();
    if (o.type === 'double') {
      ctx.beginPath();
      ctx.moveTo(ox + 28, GROUND_Y);
      ctx.lineTo(ox + 43, GROUND_Y - 28);
      ctx.lineTo(ox + 58, GROUND_Y);
      ctx.fill();
    }
  }
  ctx.shadowBlur = 0;

  // Player cube — translate to center, rotate, draw centered square
  if (alive) {
    ctx.save();
    ctx.translate(player.x, player.y);
    ctx.rotate(player.rot);
    ctx.shadowColor = '#9eff5a';
    ctx.shadowBlur = 18;
    ctx.fillStyle = '#9eff5a';
    ctx.fillRect(-player.size/2, -player.size/2, player.size, player.size);
    ctx.shadowBlur = 0;
    ctx.fillStyle = '#0a0e1a';
    ctx.fillRect(-player.size/2 + 5, -player.size/2 + 5, player.size - 10, player.size - 10);
    ctx.fillStyle = '#9eff5a';
    ctx.fillRect(-3, -3, 6, 6);
    ctx.restore();
  }

  // HUD
  ctx.textAlign = 'left';
  ctx.font = 'bold 22px JetBrains Mono, monospace';
  ctx.fillStyle = '#9eff5a';
  ctx.fillText(`SCORE  ${Math.floor(score).toString().padStart(5,'0')}`, 18, 32);
  ctx.font = '11px JetBrains Mono, monospace';
  ctx.fillStyle = '#9098b8';
  ctx.fillText(`BEST   ${Math.floor(best).toString().padStart(5,'0')}`, 18, 50);
  ctx.fillText(`SPEED  ${speed.toFixed(2)}x`, 18, 66);

  // Death overlay
  if (!alive) {
    ctx.fillStyle = `rgba(255,90,106,${0.25 + Math.sin(deathT * 0.18) * 0.08})`;
    ctx.fillRect(0, 0, W, H);
    ctx.textAlign = 'center';
    ctx.fillStyle = '#fff';
    ctx.font = 'bold 44px Bricolage Grotesque, sans-serif';
    ctx.fillText('CRASHED', W/2, H/2 - 6);
    ctx.font = '13px JetBrains Mono, monospace';
    ctx.fillStyle = '#9eff5a';
    ctx.fillText('press SPACE or click to restart', W/2, H/2 + 24);
  }

  requestAnimationFrame(loop);
}

reset();
loop();
Things to try changing:
  • Replace the cube with a circle. Hint: ctx.arc.
  • Add a double-jump. Track an airJumps counter and only allow one before hitting ground.
  • Replace the spike triangles with a different obstacle shape — saws? walls? falling blocks?
  • Make the player a different color when in the air vs. on the ground.
  • Add coins that scroll along with obstacles. Each one bumps score by 100.
