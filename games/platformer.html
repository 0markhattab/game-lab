Mini Platformer

Run, jump, collect coins, reach the flag. The simplest possible Mario. Underneath it's the most subtle of the four templates: solid-ground collision is where most beginner platformers fall apart. We do it the right way.

canvas gravity + jump solid platforms coins + goal level data
←→ or AD move SPACE or ↑ jump R restart
Open in Studio →
[01]

How it works

The level is just an array of platform rectangles, an array of coins, and a goal rectangle. The player is a rectangle too. Everything else is gravity, input, and collision response.

The key trick: we update the player's x position first and resolve any horizontal collisions. Then we update y and resolve any vertical collisions. By splitting the axes, "running into a wall" and "landing on a floor" become two separate, simple problems instead of one complicated one.

Player size
28 × 32 px
Move speed
4 px / frame
Jump impulse
vy = -13
Gravity
0.6 px/frame²
Max fall
12 px / frame
Coyote time
6 frames
Coyote time. Real platformers let you jump for a few frames after you've walked off a ledge. Without it, jumps off cliffs feel unforgiving and players blame the controls. We track coyote — frames since you were last on solid ground — and let the jump fire if it's small.

// === MINI PLATFORMER ===========================================
// Run, jump, collect coins, touch the flag.

const canvas = document.getElementById('game');
const ctx = canvas.getContext('2d');
const W = canvas.width;   // 800
const H = canvas.height;  // 450

// --- Tunable constants ---
const GRAVITY      = 0.6;
const MAX_FALL     = 12;
const MOVE_SPEED   = 4;
const JUMP_V       = -13;
const COYOTE_FRAMES = 6;

// --- Level data ---
// One ground floor + a series of floating platforms. Coins on top
// of some, a goal flag at the far right. This is just data — swap
// it out and you have a new level with no code changes.
const LEVEL = {
  platforms: [
    { x:   0, y: 410, w: 800, h: 40 },          // ground floor
    { x: 130, y: 340, w: 110, h: 14 },          // first hop
    { x: 290, y: 290, w: 110, h: 14 },
    { x: 450, y: 250, w:  90, h: 14 },
    { x: 590, y: 290, w: 110, h: 14 },
    { x: 230, y: 200, w:  80, h: 14 },          // upper route
    { x: 380, y: 150, w:  80, h: 14 },
    { x: 540, y: 110, w:  80, h: 14 },
    { x: 700, y: 170, w:  90, h: 14 },
  ],
  coins: [
    { x: 175, y: 305 }, { x: 335, y: 255 }, { x: 487, y: 215 },
    { x: 635, y: 255 }, { x: 263, y: 165 }, { x: 413, y: 115 },
    { x: 573, y:  75 }, { x: 740, y: 135 },
  ],
  goal: { x: 745, y: 110, w: 14, h: 60 },
  start: { x: 30, y: 360 }
};

let player, coins, goalReached, score, deathCount;

function reset() {
  player = {
    x: LEVEL.start.x, y: LEVEL.start.y,
    w: 28, h: 32,
    vx: 0, vy: 0,
    onGround: false,
    coyote: 0,
    facing: 1
  };
  coins = LEVEL.coins.map(c => ({ ...c, taken: false, t: Math.random() * 6 }));
  goalReached = false;
  score = 0;
}

// --- Input ---
const keys = {};
window.addEventListener('keydown', e => {
  keys[e.code] = true;
  if (['Space','ArrowUp','ArrowLeft','ArrowRight','ArrowDown'].includes(e.code)) {
    e.preventDefault();
  }
  if (e.code === 'KeyR') reset();
});
window.addEventListener('keyup', e => { keys[e.code] = false; });
document.getElementById('restart-btn').addEventListener('click', reset);

// --- AABB overlap test ---
function overlap(a, b) {
  return a.x < b.x + b.w &&
         a.x + a.w > b.x &&
         a.y < b.y + b.h &&
         a.y + a.h > b.y;
}

function update() {
  if (goalReached) return;

  // Horizontal input
  let ax = 0;
  if (keys.ArrowLeft  || keys.KeyA) { ax -= 1; player.facing = -1; }
  if (keys.ArrowRight || keys.KeyD) { ax += 1; player.facing =  1; }
  player.vx = ax * MOVE_SPEED;

  // Jump (with coyote time)
  const jumpHeld = keys.Space || keys.ArrowUp || keys.KeyW;
  if (jumpHeld && player.coyote > 0) {
    player.vy = JUMP_V;
    player.coyote = 0;
    player.onGround = false;
  }

  // Gravity
  player.vy += GRAVITY;
  if (player.vy > MAX_FALL) player.vy = MAX_FALL;

  // ============================================================
  // Move + collide on each axis SEPARATELY. This is the key idea.
  // ============================================================

  // X axis
  player.x += player.vx;
  for (const p of LEVEL.platforms) {
    if (overlap(player, p)) {
      // Push out along x by the smaller side
      if (player.vx > 0) player.x = p.x - player.w;
      else if (player.vx < 0) player.x = p.x + p.w;
      player.vx = 0;
    }
  }

  // Y axis
  player.y += player.vy;
  player.onGround = false;
  for (const p of LEVEL.platforms) {
    if (overlap(player, p)) {
      if (player.vy > 0) {
        player.y = p.y - player.h;
        player.onGround = true;
      } else if (player.vy < 0) {
        player.y = p.y + p.h;
      }
      player.vy = 0;
    }
  }

  // Coyote time: while grounded, refresh window. Otherwise count down.
  if (player.onGround) player.coyote = COYOTE_FRAMES;
  else if (player.coyote > 0) player.coyote--;

  // World bounds
  if (player.x < 0) player.x = 0;
  if (player.x + player.w > W) player.x = W - player.w;

  // Fell off the bottom
  if (player.y > H + 50) reset();

  // Coins
  for (const c of coins) {
    if (c.taken) continue;
    c.t += 0.12;
    const cb = { x: c.x - 10, y: c.y - 10, w: 20, h: 20 };
    if (overlap(player, cb)) {
      c.taken = true;
      score += 10;
    }
  }

  // Goal
  if (overlap(player, LEVEL.goal)) goalReached = true;
}

function draw() {
  // Background
  const bg = ctx.createLinearGradient(0, 0, 0, H);
  bg.addColorStop(0, '#1a2340');
  bg.addColorStop(1, '#0a0e1a');
  ctx.fillStyle = bg;
  ctx.fillRect(0, 0, W, H);

  // Grid
  ctx.strokeStyle = 'rgba(35,44,74,0.5)';
  ctx.lineWidth = 1;
  for (let x = 0; x < W; x += 32) {
    ctx.beginPath(); ctx.moveTo(x, 0); ctx.lineTo(x, H); ctx.stroke();
  }
  for (let y = 0; y < H; y += 32) {
    ctx.beginPath(); ctx.moveTo(0, y); ctx.lineTo(W, y); ctx.stroke();
  }

  // Platforms
  for (const p of LEVEL.platforms) {
    ctx.fillStyle = '#1d2540';
    ctx.fillRect(p.x, p.y, p.w, p.h);
    // top highlight
    ctx.fillStyle = '#9eff5a';
    ctx.fillRect(p.x, p.y, p.w, 2);
  }

  // Coins (spinning ellipses)
  for (const c of coins) {
    if (c.taken) continue;
    const sx = Math.abs(Math.cos(c.t)) * 8 + 2;
    ctx.fillStyle = '#ffb347';
    ctx.shadowColor = '#ffb347';
    ctx.shadowBlur = 12;
    ctx.beginPath();
    ctx.ellipse(c.x, c.y, sx, 8, 0, 0, Math.PI * 2);
    ctx.fill();
    ctx.shadowBlur = 0;
  }

  // Goal flag
  const g = LEVEL.goal;
  ctx.fillStyle = '#5a6388';
  ctx.fillRect(g.x + g.w/2 - 1, g.y, 2, g.h);
  ctx.fillStyle = '#ff3eb5';
  ctx.shadowColor = '#ff3eb5';
  ctx.shadowBlur = 10;
  ctx.beginPath();
  ctx.moveTo(g.x + g.w/2, g.y);
  ctx.lineTo(g.x + g.w/2 + 24, g.y + 8);
  ctx.lineTo(g.x + g.w/2, g.y + 16);
  ctx.fill();
  ctx.shadowBlur = 0;

  // Player
  ctx.save();
  ctx.translate(player.x + player.w/2, player.y + player.h/2);
  ctx.scale(player.facing, 1);
  ctx.fillStyle = '#9eff5a';
  ctx.shadowColor = '#9eff5a';
  ctx.shadowBlur = 14;
  ctx.fillRect(-player.w/2, -player.h/2, player.w, player.h);
  ctx.shadowBlur = 0;
  // eye
  ctx.fillStyle = '#0a0e1a';
  ctx.fillRect(player.w/2 - 10, -player.h/2 + 8, 5, 5);
  // mouth (subtle)
  ctx.fillRect(player.w/2 - 14, -player.h/2 + 18, 8, 2);
  ctx.restore();

  // HUD
  ctx.font = 'bold 18px JetBrains Mono, monospace';
  ctx.fillStyle = '#9eff5a';
  ctx.textAlign = 'left';
  ctx.fillText(`COINS  ${score / 10} / ${coins.length}`, 14, 28);
  ctx.fillText(`SCORE  ${score}`, 14, 50);

  // Goal screen
  if (goalReached) {
    ctx.fillStyle = 'rgba(158,255,90,0.18)';
    ctx.fillRect(0, 0, W, H);
    ctx.textAlign = 'center';
    ctx.fillStyle = '#9eff5a';
    ctx.font = 'bold 48px Bricolage Grotesque, sans-serif';
    ctx.fillText('LEVEL COMPLETE', W/2, H/2 - 8);
    ctx.font = '13px JetBrains Mono, monospace';
    ctx.fillStyle = '#fff';
    ctx.fillText(`coins: ${score / 10} / ${coins.length}    score: ${score}`, W/2, H/2 + 22);
    ctx.fillText('press R to play again', W/2, H/2 + 42);
  }
}

function loop() {
  update();
  draw();
  requestAnimationFrame(loop);
}

reset();
loop();
Things to try changing:
  • Add a moving platform. Give a platform a vx and update its position each frame. Make sure the player rides it.
  • Build a second level. Copy the LEVEL object, edit the platforms, swap them on goal-reach.
  • Add an enemy that walks back and forth on a platform. Player loses if they touch it from the side.
  • Add a stomp mechanic: if the player is falling AND lands on top of an enemy, the enemy dies.
  • Replace the rectangle player with a sprite from an image. Use ctx.drawImage.
