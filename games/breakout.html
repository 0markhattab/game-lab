Breakout

A ball, a paddle, and a wall of bricks to dismantle. Atari shipped this in 1976 and the formula still holds: aim with your paddle's angle, watch the ball ricochet, and chase the satisfaction of clearing the board.

How it works

The ball has a velocity — pixels per frame in x and y. Every frame we add velocity to position, then check for collisions: walls (flip the relevant velocity), the paddle (flip vy and steer based on where it hit), and bricks (flip vy or vx depending on which side the ball entered from).

The trick that makes Breakout feel good is the paddle steering. A flat paddle bounces the ball straight up forever and the game gets boring. We make the paddle act like a slight curve: hit the left edge, ball goes left; hit the right edge, ball goes right. Same paddle, totally different game.

Bricks
5 rows × 9 cols
Brick HP
1 hit per brick
Ball speed
~5 px / frame
Paddle steer
based on hit offset
Lives
3 per game
Win condition
all bricks broken
Spatial reasoning beats hit detection. When the ball overlaps a brick, the naive thing is to flip vy and call it done. But what if the ball came in from the side? We figure out where the ball was before it overlapped the brick, and pick the bounce direction from that.

// === BREAKOUT ==================================================
// Paddle. Ball. Wall of bricks. Don't drop the ball.

const canvas = document.getElementById('game');
const ctx = canvas.getContext('2d');
const W = canvas.width;   // 640
const H = canvas.height;  // 480

// --- Brick layout ---
const BRICK_COLS = 9;
const BRICK_ROWS = 5;
const BRICK_W    = 60;
const BRICK_H    = 18;
const BRICK_PAD  = 4;
const BRICK_TOP  = 60;
const BRICK_LEFT = (W - (BRICK_COLS * (BRICK_W + BRICK_PAD) - BRICK_PAD)) / 2;

// Each row gets a different color so the player feels progress
const ROW_COLORS = ['#ff3eb5', '#ff3eb5', '#ffb347', '#5ce1ff', '#9eff5a'];

let paddle, ball, bricks, score, lives, alive, won, launched;

function reset() {
  paddle = { x: W/2 - 50, y: H - 30, w: 100, h: 12, speed: 8 };
  ball   = { x: W/2, y: H - 50, vx: 0, vy: 0, r: 7 };
  bricks = [];
  for (let r = 0; r < BRICK_ROWS; r++) {
    for (let c = 0; c < BRICK_COLS; c++) {
      bricks.push({
        x: BRICK_LEFT + c * (BRICK_W + BRICK_PAD),
        y: BRICK_TOP  + r * (BRICK_H + BRICK_PAD),
        w: BRICK_W, h: BRICK_H,
        color: ROW_COLORS[r],
        alive: true
      });
    }
  }
  score = 0;
  lives = 3;
  alive = true;
  won   = false;
  launched = false;
}

function launchBall() {
  // Slight random angle so each launch feels different
  const a = -Math.PI/2 + (Math.random() - 0.5) * 0.6;
  const sp = 5;
  ball.vx = Math.cos(a) * sp;
  ball.vy = Math.sin(a) * sp;
  launched = true;
}

// --- Input ---
const keys = {};
window.addEventListener('keydown', e => {
  keys[e.code] = true;
  if (e.code === 'Space') {
    e.preventDefault();
    if (!alive || won) reset();
    else if (!launched) launchBall();
  }
});
window.addEventListener('keyup', e => { keys[e.code] = false; });
canvas.addEventListener('mousemove', e => {
  const rect = canvas.getBoundingClientRect();
  const mx = (e.clientX - rect.left) * (W / rect.width);
  paddle.x = mx - paddle.w/2;
});
canvas.addEventListener('click', () => {
  if (!alive || won) reset();
  else if (!launched) launchBall();
});
document.getElementById('restart-btn').addEventListener('click', reset);

// --- AABB vs circle intersection. Returns the side the ball hit. ---
// We detect by checking which axis we'd have to push the ball back
// the most along to NOT be overlapping. That axis is the side we hit.
function brickHit(b, ball) {
  // Closest point on brick to ball center
  const cx = Math.max(b.x, Math.min(ball.x, b.x + b.w));
  const cy = Math.max(b.y, Math.min(ball.y, b.y + b.h));
  const dx = ball.x - cx;
  const dy = ball.y - cy;
  if (dx*dx + dy*dy > ball.r * ball.r) return null;

  // Decide bounce axis based on where the ball is relative to brick
  const overlapX = ball.r - Math.abs(dx);
  const overlapY = ball.r - Math.abs(dy);
  if (overlapX < overlapY) return 'x';   // hit left/right side
  return 'y';                            // hit top/bottom
}

function update() {
  if (!alive || won) return;

  // Paddle keyboard control
  if (keys.ArrowLeft  || keys.KeyA) paddle.x -= paddle.speed;
  if (keys.ArrowRight || keys.KeyD) paddle.x += paddle.speed;
  paddle.x = Math.max(0, Math.min(W - paddle.w, paddle.x));

  // Before launch: ball sticks to top of paddle
  if (!launched) {
    ball.x = paddle.x + paddle.w/2;
    ball.y = paddle.y - ball.r - 2;
    return;
  }

  // Move ball
  ball.x += ball.vx;
  ball.y += ball.vy;

  // Wall collisions
  if (ball.x - ball.r < 0)         { ball.x = ball.r;       ball.vx *= -1; }
  if (ball.x + ball.r > W)         { ball.x = W - ball.r;   ball.vx *= -1; }
  if (ball.y - ball.r < 0)         { ball.y = ball.r;       ball.vy *= -1; }

  // Floor — lose a life
  if (ball.y - ball.r > H) {
    lives--;
    if (lives <= 0) { alive = false; }
    else { launched = false; }
    return;
  }

  // Paddle
  if (ball.y + ball.r > paddle.y &&
      ball.y - ball.r < paddle.y + paddle.h &&
      ball.x > paddle.x && ball.x < paddle.x + paddle.w &&
      ball.vy > 0) {
    // Hit position from -1 (left edge) to +1 (right edge)
    const hit = (ball.x - (paddle.x + paddle.w/2)) / (paddle.w/2);
    const angle = hit * (Math.PI / 3);  // up to 60° steering
    const speed = Math.hypot(ball.vx, ball.vy);
    ball.vx = Math.sin(angle) * speed;
    ball.vy = -Math.abs(Math.cos(angle) * speed);
    ball.y = paddle.y - ball.r;
  }

  // Bricks
  for (const b of bricks) {
    if (!b.alive) continue;
    const side = brickHit(b, ball);
    if (side) {
      b.alive = false;
      score += 10;
      if (side === 'x') ball.vx *= -1;
      else              ball.vy *= -1;
      // Speed up slightly each break, capped
      const sp = Math.min(8, Math.hypot(ball.vx, ball.vy) + 0.05);
      const cur = Math.hypot(ball.vx, ball.vy);
      ball.vx *= sp / cur;
      ball.vy *= sp / cur;
      break; // only one brick per frame
    }
  }

  // Win condition
  if (bricks.every(b => !b.alive)) won = true;
}

function draw() {
  ctx.fillStyle = '#0a0e1a';
  ctx.fillRect(0, 0, W, H);

  // Subtle grid backdrop
  ctx.strokeStyle = 'rgba(35,44,74,0.5)';
  for (let x = 0; x < W; x += 32) {
    ctx.beginPath(); ctx.moveTo(x, 0); ctx.lineTo(x, H); ctx.stroke();
  }

  // Bricks (with glow)
  for (const b of bricks) {
    if (!b.alive) continue;
    ctx.fillStyle = b.color;
    ctx.shadowColor = b.color;
    ctx.shadowBlur = 6;
    ctx.fillRect(b.x, b.y, b.w, b.h);
    // inner highlight
    ctx.shadowBlur = 0;
    ctx.fillStyle = 'rgba(255,255,255,0.15)';
    ctx.fillRect(b.x, b.y, b.w, 3);
  }

  // Paddle
  ctx.fillStyle = '#9eff5a';
  ctx.shadowColor = '#9eff5a';
  ctx.shadowBlur = 12;
  ctx.fillRect(paddle.x, paddle.y, paddle.w, paddle.h);
  ctx.shadowBlur = 0;

  // Ball
  ctx.fillStyle = '#fff';
  ctx.shadowColor = '#fff';
  ctx.shadowBlur = 14;
  ctx.beginPath();
  ctx.arc(ball.x, ball.y, ball.r, 0, Math.PI * 2);
  ctx.fill();
  ctx.shadowBlur = 0;

  // HUD
  ctx.font = 'bold 18px JetBrains Mono, monospace';
  ctx.fillStyle = '#9eff5a';
  ctx.textAlign = 'left';
  ctx.fillText(`SCORE ${score}`, 14, 28);
  ctx.textAlign = 'right';
  ctx.fillText(`LIVES ${'●'.repeat(lives)}`, W - 14, 28);

  if (!launched && alive && !won) {
    ctx.textAlign = 'center';
    ctx.font = '12px JetBrains Mono, monospace';
    ctx.fillStyle = '#9eff5a';
    ctx.fillText('press SPACE or click to launch', W/2, H - 50);
  }

  // Game over / win
  if (!alive) {
    ctx.fillStyle = 'rgba(255,90,106,0.25)';
    ctx.fillRect(0, 0, W, H);
    ctx.textAlign = 'center';
    ctx.fillStyle = '#fff';
    ctx.font = 'bold 42px Bricolage Grotesque, sans-serif';
    ctx.fillText('GAME OVER', W/2, H/2 - 8);
    ctx.font = '12px JetBrains Mono, monospace';
    ctx.fillStyle = '#9eff5a';
    ctx.fillText('SPACE to play again', W/2, H/2 + 22);
  }
  if (won) {
    ctx.fillStyle = 'rgba(158,255,90,0.18)';
    ctx.fillRect(0, 0, W, H);
    ctx.textAlign = 'center';
    ctx.fillStyle = '#9eff5a';
    ctx.font = 'bold 42px Bricolage Grotesque, sans-serif';
    ctx.fillText('CLEARED!', W/2, H/2 - 8);
    ctx.font = '12px JetBrains Mono, monospace';
    ctx.fillStyle = '#fff';
    ctx.fillText(`final score ${score} — SPACE to play again`, W/2, H/2 + 22);
  }
}

function loop() {
  update();
  draw();
  requestAnimationFrame(loop);
}

reset();
loop();
